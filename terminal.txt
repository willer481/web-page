# Step 1: Initialize Git in your project folder
git init

# Step 2: Add all files to staging
git add .

# Step 3: Commit the files
git commit -m "Initial commit"

# Step 4: Add your GitHub repo as the remote
# IMPORTANT: Replace [your repo URL] with your actual GitHub repo URL
git remote add origin https://github.com/willer481/web-page.git

# Step 5: Push your code to GitHub
git push -u origin main
